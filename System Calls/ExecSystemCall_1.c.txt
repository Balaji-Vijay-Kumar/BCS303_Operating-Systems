#include <stdio.h>

int main() {
    printf("Hello, Memory image replacement successful\n");

    int a, b;
    printf("Enter the values of A and B\n");
    scanf("%d %d", &a, &b);

    int sum = a + b;
    printf("The Sum of A and B is %d\n", sum);

    return 0;
}
